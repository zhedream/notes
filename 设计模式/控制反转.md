控制什么?

```js
function sum() {
  // 注意: 这里的依赖写死了!
  const a = 1; // 依赖 a
  const b = 2; // 依赖 b
  return a + b;
}

function sum(a, b) {
  return a + b;
}
sum(3, 4); // 这里就有点 依赖注入的感觉了, 依赖 a 和 依赖 b , 没有写死 是在程序 运行时 被注入了
```
