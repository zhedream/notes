```js
// 获取 ['','']
const paths = data.paths;
let detail = data.detail;
let names = [];
let deep = 0;
const getName = (tree) => {
  tree.forEach((item) => {
    if (item.value == paths[deep]) {
      deep++;
      names.push(item.label);
    }
    if (item.children && item.children.length > 0) getName(item.children);
  });
};
getName(detail);
console.log(data.code, names);
```
