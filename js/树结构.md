function getTreebyKey(tree, value) {
  let tem;
  const get = function (tree, value) {
    for (let i = 0; i < tree.length; i++) {
      const item = tree[i];
      if (item.value == value) {
        tem = item;
        break;
      }
      if (item.children && item.children.length > 0) {
        arguments.callee(item.children, value);
      }
    }
  };
  get(tree, value);
  return tem;
}
